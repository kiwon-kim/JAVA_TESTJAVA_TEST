라즈베리파이 헬로월드

#include <stdio.h>
#include <stdlib.h>
#include <errno.h>

#include <wiringPi.h>
#include <wiringSerial.h>

char device [] = "/dev/ttyACM0";
int fd;
unsigned long baud = 9600;

int main (void)
{
        printf("%s \n","Raspberry");
        fflush(stdout);

        if((fd= serialOpen(device, baud)) < 0)
                exit(1);

        if(wiringPiSetupGpio() == -1)
                return 1;

        while(1)
        {
                if(serialDataAvail(fd))
                {
                        char newChar = serialGetchar(fd);
                        printf("%c",newChar);
                        fflush(stdout);
                }
        }
        return 0;
}
